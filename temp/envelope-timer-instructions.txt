# envelope-accrue.service
# Place at: /etc/systemd/system/envelope-accrue.service
#
# Runs the monthly accrual script on the 1st of each month.
# Adds configured monthly_amount to each sinking_fund and budget envelope.

[Unit]
Description=Monthly envelope accrual for hledger envelope system
After=network.target

[Service]
Type=oneshot
User=server
WorkingDirectory=/home/server/git_projects/hledger-wrapper
ExecStart=/home/server/git_projects/hledger-wrapper/api/venv/bin/python /home/server/git_projects/hledger-wrapper/api/accrue.py
StandardOutput=journal
StandardError=journal

---

# envelope-accrue.timer
# Place at: /etc/systemd/system/envelope-accrue.timer

[Unit]
Description=Run envelope accrual on the 1st of each month

[Timer]
OnCalendar=*-*-01 06:00:00
Persistent=true

[Install]
WantedBy=timers.target

---

# To install:
#   sudo cp envelope-accrue.service /etc/systemd/system/
#   sudo cp envelope-accrue.timer   /etc/systemd/system/
#   sudo systemctl daemon-reload
#   sudo systemctl enable --now envelope-accrue.timer
#
# To check:
#   systemctl list-timers envelope-accrue.timer
#   journalctl -u envelope-accrue.service
